using System;
using System.Collections.Generic;

namespace Deveel.Data.Sql.Compile {
	/// <summary>
	/// Represents a keyword found during the compilation of a source text.
	/// </summary>
	/// <seealso cref="ISqlNode"/>
	[Serializable]
	public sealed class SqlKeyNode : ISqlChildNode {
		internal SqlKeyNode(Token token) {
			this.token = token;
			Text = token.Text;
		}

		private readonly Token token;
		private ISqlNode parent;

		string ISqlNode.NodeName {
			get { return Text; }
		}

		ISqlNode ISqlNode.Parent {
			get { return parent; }
		}

		IEnumerable<ISqlNode> ISqlNode.ChildNodes {
			get { return new ISqlNode[0]; }
		}

		IEnumerable<Token> ISqlNode.Tokens {
			get { return new Token[] {token}; }
		}

		void ISqlChildNode.SetParent(ISqlNode node) {
			parent = node;
		}

		/// <summary>
		/// Gets the source text of the keyword.
		/// </summary>
		public string Text { get; private set; }
	}
}
